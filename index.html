Example: If Statements

if_example.emo
mood a = 10
mood b = 20
if a < b
    share("a is less than b")
end

if a == b
    share("a is equal to b")
end

Example: While Loops (Endure)

while_example.emo
mood counter = 0
endure counter < 5
    share("Counter is:")
    share(counter)
    mood counter = counter + 1
end
share("Endure loop finished")

Example: For Loops (Persevere)

for_example.emo
persevere i = 1 to 5
    share("i in persevere loop:")
    share(i)
end

persevere j = 0 to 10 step 2
    share("j in persevere loop with step:")
    share(j)
end
share("Persevere loop finished")
